Fix direction: pages are sorted by date desc, and within a series the chapter number increases with date. PaperMod's default wires these backwards. */ -}} {{- $pages := where site.RegularPages "Section" .Section }} {{- if and (gt (len $pages) 1) (in $pages . ) }}

{{- .Name -}} {{- end }} {{- with $pages.Next . }} {{ i18n "next_page" }} »
{{- .Name -}} {{- end }}
